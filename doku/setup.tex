\section{Setup-Anleitung}

\begin{enumerate} 
	\item mkdir dvdarchiv \&\& cd dvdarchiv
	\item git clone \emph{repo-link} .
	\item git checkout jasmin
	\item mvn install
	\item Projekt in IDE öffnen und runnen
	\item localhost:8080 im Browser öffnen
	\item Bier öffnen!
\end{enumerate}
